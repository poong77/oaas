#결정근거효과
1 미들웨어 미사용 (admin)/(user) layout이 이미 force-dynamic. 루트 layout도 EmergencyBanner DB 쿼리로 어차피 dynamic. NextAuth 충돌 위험 0, 빌드 복잡도 0, edge runtime 제약 0
2 CSS 변수 cascade [data-role]으로 --color-brand-* 오버라이드. className="bg-brand-700"은 자동으로 다른 색 렌더. 기존 50+ 파일 무수정 (회귀 위험 ↓)
3 RoleScope 단일 진입점 모든 분기(mode 결정, 호텔리어 UI 조건부 노출, ViewModeBanner)를 단일 서버 컴포넌트에 통합. 로직 분산 없음. 어드민 모드에서 EmergencyBanner DB 쿼리 자체가 발생 안 함 (성능 ↑)
4 viewMode 쿠키 (UI 표시용) HttpOnly=false, SameSite=Lax, MaxAge=4h. 권한 체크는 항상 user.role 기준. 위조해도 무해 (서버 권한 절대 우회 불가)
5 자물쇠 메뉴 = button disabled Link 사용 금지. 매니저가 admin-only 메뉴 클릭해도 404 절대 발생 안 함. design.md E6 위험 사전 차단
6 useViewMode mounted 패턴 첫 렌더 항상 false → 서버 결정 결과와 일치. 마운트 후 실제 쿠키값 반영. SSR/CSR hydration mismatch 차단

resolveRoleMode() — UI 모드 결정 로직

export function resolveRoleMode( userRole: UserRole | null, viewModeCookie: string | undefined, ): RoleMode { if (!userRole) return 'hotelier'; // 비로그인 = 호텔리어 톤 if (userRole === 'hotelier') return 'hotelier'; // 본인은 항상 자기 모드 if (viewModeCookie === 'hotelier') return 'hotelier'; // manager/admin 시점 보기 return userRole; // 평소엔 manager/admin }

CSS 변수 cascade — globals.css

@theme { --color-brand-700: oklch(0.457 0.240 277.02); /* 어드민 indigo */ } [data-role='manager'] { --color-brand-700: #1d4ed8; } /* 매니저 cobalt */ [data-role='hotelier'] { --color-brand-700: #15803d; } /* 호텔리어 green */ /* 기존 코드 무수정 */ <button className="bg-brand-700"> → CSS 변수만 다르게 해석 → 자동 분기

RoleScope — 모든 분기의 단일 진입점

export async function RoleScope({ children }) { const session = await auth(); const cookieStore = await cookies(); const viewModeCookie = cookieStore.get(VIEW_MODE_COOKIE)?.value; const userRole = session?.user?.role ?? null; const mode = resolveRoleMode(userRole, viewModeCookie); const showHotelierUi = mode === 'hotelier'; return ( <div data-role={mode} className="contents"> {viewModeActive && <ViewModeBanner userRole={userRole} />} {showHotelierUi && <EmergencyBanner />} {showHotelierUi && <Header />} <main>{children}</main> {showHotelierUi && <ChatbotFab embedUrl={chatbotEmbedUrl} />} </div> ); }
